(function(module) {
  'use strict';
  module.filter('gridObj', gridObj);
 
  function gridObj($filter) {
 
    return function(input, type) {
      var dateFormat = 'MMM/dd/yy hh:mm';
      var output = input;
 
      if (input) {
 
        switch (type) {
 
          case 'OBJECT' :
            output = input.displayName;
            break;
          case 'DATE' :
            output = $filter('date')(input, dateFormat);
            break;
        }
      }
 
      return output;
    };
  }
})(angular.module('web-app'));
