(function(module){
  'use strict';
  module.factory('filtersMapperSrv', filtersMapperSrv);
 
  function filtersMapperSrv(filterCnst) {
 
    function mapData(rawData){
      var result = {};
      if (rawData && rawData.data){
        result = _.chain(rawData.data)
            .groupBy(function(item){
              item.componentType = getComponentType(item);
              return (item.group.slice(0, item.group.indexOf('.')));
            })
            .value();
      }
      return result;
    }
 
    function getComponentType(entity) {
      if (entity) {
        switch (entity.type) {
          case 'TEXT':
            if (entity.validValues.length) {
              return filterCnst.TYPE.checkBox;
            } else if (!entity.validValues.length) {
              return filterCnst.TYPE.autoComplete;
            }
            break;
 
        }
      }
    }
    return {
      mapData: mapData
    };
  }
 
})(angular.module('web-app'));
